var main = () => {
  const getStright = (a, rogue = 0) => {
    let stright = true;
    // console.log(JSON.stringify([...a]));
    // 去重
    if (Array.from(new Set(a)).length !== a.length) {
      return false;
    }
    a = a.sort((a, b) => a - b);
    let len = a.length;
    for (let i = 0; i < len; i++) {
      if (a[i + 1]) {
        if (a[i] === a[i + 1] - 1) { // 1
          continue;
        } else if (a[i + 1] - a[i] > rogue + 1) { // 赖子填补
          stright = false;
          break;
        } else if (rogue > 0){
          a.splice(i + 1, 0, a[i] + 1);
          rogue--;
          len++;
        }
      }
      if (rogue < 0) {
        stright = false;
        break;
      }
    }
    return stright;
  }
  const getRandom = () => {
    const getReturnVal = (arr) => {
      const returnVal = {
        arr: [],
        rogue: 0,
      };
      let len = 5;
      for (let i = 1; i <= len; i++) {
        const num = Math.floor(Math.random() * 54 + 1);
        if (Object.prototype.toString.call(arr[num]) === '[object Number]') {
          returnVal.arr.push(arr[num]);
        } else if (returnVal.rogue < 2) {
          returnVal.rogue++;
        }
      }
      return  returnVal;
    }
    const initEmpty = () => {
      const arr = [];
      for (let i = 1; i <= 13; i++) {
        for (let j = 0; j < 4; j++) {
          arr.push(i);
        }
      }
      arr.push('a');
      arr.push('b');
      return arr;
    }
    return getReturnVal(initEmpty());
  }
  for (let i = 0; i < 500; i++) {
    const random = getRandom();
    const isStright = getStright(random.arr, random.rogue);
    if (isStright) {
      console.log('顺子', JSON.stringify(random.arr));
    }
  }
}
main();

console.log(`

题：
  从扑克牌中随机抽5 张牌，判断是不是一个顺子，即这5 张牌是不是连续的。2-10 为数字本身，A 为1，J 为11，Q 为12，K 为13，而大小王可以看成任意数字。

`);

// 扑克牌
const poker = [];
// 单色扑克数组
const pokerList = [];

// 小王
const leper = 53;
// 大王
const king = 54;

// 花色
const pokerColors = ['黑桃', '红桃', '梅花', '方块'];

// 抽取的牌数
const extractNum = 5;

// 抽取的牌
let extractPoker = {};

// 构建单色13张
function buildPokerList() {
  for (let i = 1; i <= 13; i += 1) {
    pokerList.push(i);
  }
}

// 构建54张扑克
function buildPoker() {
  buildPokerList();
  pokerColors.forEach(color => {
    pokerList.forEach(num => {
      poker.push({
        color,
        num
      });
    });
  });
  poker.push({
    color: '小王',
    num: leper
  });
  poker.push({
    color: '大王',
    num: king
  });
}

// 抽取需要的牌
function RandomSelection() {
  if (Object.keys(extractPoker).length === extractNum) return;
  else {
    const idx = getRndInteger();
    if (!extractPoker[idx]) {
      extractPoker[idx] = poker[idx];
    }
    RandomSelection();
  }
}

// 获取随机数
function getRndInteger() {
  const min = 0;
  const max = 53;
  return Math.floor(Math.random() * (max - min + 1) + min);
}

// 去重
function unique() {
  return [...new Set(Object.values(extractPoker).map(o => o.num))].length;
}

// 头部去空
function trim(list) {
  if (list[0]) {
    return list;
  } else {
    list.shift();
    return trim(list);
  }
}

// 运算
function operation() {
  debugger
  const uniqueExtractPokerLength = unique();
  if (uniqueExtractPokerLength < extractNum) {
    console.log('有相同的数字');
    return;
  }
  // 已牌数字大小做下标，自动排序手牌
  const list = new Array();
  const lepers = [];
  Object.values(extractPoker).forEach(poke => {
    if (poke.num === leper || poke.num === king) {
      // 抽出癞子
      lepers.push(poke);
    } else {
      list[poke.num] = poke;
    }
  });
  trim(list);

  // 如果手牌长度 - 非癞子牌数 <= 癞子总数量       就是一个顺子
  if (list.length - (extractNum - lepers.length) <= lepers.length) {
    console.log(`
    --------------------这是一个顺子--------------------
    ${Object.values(extractPoker).map(o => JSON.stringify(o).replace(/"/g, ''))}
    --------------------------------------------------
    `);
  } else {
    console.log('不是顺子');
  }
}

function main() {
  buildPoker();
  // RandomSelection(5);
  extractPoker = {};
  extractPoker['0'] = {
    num: 6
  };
  extractPoker['3'] = {
    num: 7
  };
  extractPoker['6'] = {
    num: 8
  };
  extractPoker['9'] = {
    num: 54
  };
  extractPoker['10'] = {
    num: 53
  };
  console.log('随机抽取：', ...Object.values(extractPoker));

  operation();
}

main();
